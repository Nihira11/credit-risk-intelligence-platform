Calculate IV (Simple Manual Method)

# manual IV calculation
calc_iv <- function(data, target_col, feature_col) {
  tbl <- table(data[[feature_col]], data[[target_col]])
  good_total <- sum(tbl[, 1], na.rm = TRUE)
  bad_total <- sum(tbl[, 2], na.rm = TRUE)
  if (good_total == 0 || bad_total == 0) return(0)
  iv <- 0
  for (i in 1:nrow(tbl)) {
    good_pct <- tbl[i, 1] / good_total
    bad_pct <- tbl[i, 2] / bad_total
    if (good_pct > 0 && bad_pct > 0) {
      woe <- log(good_pct / bad_pct)
      iv <- iv + (good_pct - bad_pct) * woe
    }
  }
  return(abs(iv))
}
